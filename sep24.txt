public EventModel getEvent(long eID){
        SQLiteDatabase db = this.getReadableDatabase();

        String selectQuery = "SELECT * FROM" +TABLE2_NAME+ "WHERE" +col1+ "=" +eID;

        Log.e(LOG, selectQuery);

        Cursor c = db.rawQuery(selectQuery, null);
        if(c != null)
            c.moveToFirst();

        EventModel event = new EventModel(int eventID, String event, String eventDe, String eventLocation, String sTime, String eTime);
        event.setEventID(c.getInt(c.getColumnIndex(col1)));
        event.setEventName(c.getString(c.getColumnIndex(col2)));
        event.setEventDescription(c.getString(c.getColumnIndex(col3)));
        event.setEventLocation(c.getString(c.getColumnIndex(col4)));

        return event;
    }
